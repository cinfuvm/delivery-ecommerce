La base de datos modela pedidos, productos, usuarios y estados de entrega.

[source,sql]
----
CREATE TABLE users (
  id INTEGER PRIMARY KEY,
  username TEXT NOT NULL,
  password TEXT NOT NULL
);

CREATE TABLE delivery_orders (
  id INTEGER PRIMARY KEY,
  user_id INTEGER,
  product TEXT,
  status TEXT,
  created_at TEXT,
  FOREIGN KEY(user_id) REFERENCES users(id)
);
----
